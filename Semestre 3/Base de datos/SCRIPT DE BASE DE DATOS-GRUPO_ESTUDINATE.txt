CREATE  TABLE   asignatura(
	
	codigo int NOT NULL,
	nombre varchar (15) NOT NULL,
	creditos int NOT NULL,	
	
	PRIMARY KEY(codigo)
); 

CREATE TABLE grupo(
	cod_asig int NOT NULL,
	codigo int NOT NULL,
	cupo int NOT NULL,

	PRIMARY KEY (cod_asig,codigo),
	FOREIGN KEY(cod_asig)
	REFERENCES asignatura (codigo) 
);

CREATE TABLE estudiante( 
	codigo int NOT NULL,
	nombre varchar(40) NOT NULL,
	fercha_nto date NOT NULL,

	PRIMARY KEY (codigo)
);
CREATE TABLE estudiante_grupo( 
	cod_estudiante int NOT NULL, 
	cod_asig int NOT NULL, 
	cod_grupo int NOT NULL,

	PRIMARY KEY (cod_estudiante, cod_asig,cod_grupo),

	FOREIGN KEY (cod_estudiante)

		REFERENCES estudiante(codigo),
	FOREIGN KEY (cod_grupo,cod_asig)	

		REFERENCES grupo(codigo,cod_asig)
		
);
